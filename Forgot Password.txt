    setLoading(true)
    const { error: resetError } = await supabase.auth.resetPasswordForEmail(email, {
      redirectTo: `${window.location.origin}/auth/callback?next=/dashboard/profile`,
    })

    if (resetError) {
      if (resetError.status === 429 || resetError.message.toLowerCase().includes('rate limit')) {
        toast.error('Limit email gratis tercapai (Sistem sibuk). Coba gunakan login via Google jika pernah terhubung, atau tunggu beberapa saat.')
      } else {
        toast.error('Gagal mengirim link reset: ' + resetError.message)
      }
    } else {
      toast.success('Link reset ajaib telah dikirim! Cek inbox/spam emailmu.')
    setShowForgotPassword(false)
    }
    setLoading(false)
  }
